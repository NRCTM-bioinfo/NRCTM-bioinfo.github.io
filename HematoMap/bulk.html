Chapter 5 Bulk RNA-seq

5.1 Load bulk RNA-seq data

# Library package
library(HematoMap)

# Read TPM matrix of B-ALL
ball.tpm <- readRDS("data/BALL_bulk_TPM.rds")
dim(ball.tpm)
## [1] 59982    10
head(ball.tpm)
hmap.bulk.ball <- computeLassoScore(ball.tpm)

# Read TPM matrix of AML
aml.tpm <- readRDS("data/AML_bulk_TPM.rds")
dim(aml.tpm)
## [1] 59344    10
head(aml.tpm)
hmap.bulk.aml <- computeLassoScore(aml.tpm)

5.2 Visualization

plotLassoLollipop(hmap.bulk.ball)

plotLassoLollipop(hmap.bulk.aml)

plotLassoTree(hmap.bulk.ball, color.mapping = "value", 
              title = "B-ALL bulk", point.size = 2, 
              label.cell = T)

plotLassoTree(hmap.bulk.ball, color.mapping = "cell.type", 
              title = "B-ALL bulk", point.size = 2, 
              label.cell = T)

plotLassoTree(hmap.bulk.aml, color.mapping = "value", 
              title = "AML bulk", point.size = 2, 
              label.cell = T)

plotLassoTree(hmap.bulk.aml, color.mapping = "cell.type", 
              title = "AML bulk", point.size = 2, 
              label.cell = T)
