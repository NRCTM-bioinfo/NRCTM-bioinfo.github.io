5.1 Calculate WT-up genes

# DEGs between p53 mut (top mutations) and p53 wt 

enroll.data <- tcga.mutation.tp53[tcga.mutation.tp53$Amino_Acid_Change %in% tp53.mut.topall$Mutation, ]


total.deg <- NULL
for (kkk in 1:length(TCGA.ALL)) {
  pdata <- tcga.meta
  sum(pdata$sample != colnames(tcga.fpkm))
  
  pdata$contrast <- "other"
  pdata$contrast[which(pdata$TP53mut == "Yes" & pdata$project_id == TCGA.ALL[kkk] & pdata$sample %in% enroll.data$Sample_ID)] <- "c1"
  pdata$contrast[which(pdata$TP53mut == "No" & pdata$project_id == TCGA.ALL[kkk] & pdata$WithMutation == "Yes")] <- "c2"
  
  message(kkk, "  ", TCGA.ALL[kkk], "   ", sum(pdata$contrast %in% c("c1")), "   ", sum(pdata$contrast %in% c("c2")) )
  
  if (length(unique(pdata$contrast)) == 3 & sum(table(pdata$contrast) < 1) == 0 ) {
    pdata.sub <- pdata[which(pdata$contrast != "other"), ]
    design <- model.matrix(~ 0 + as.factor(contrast), data = pdata.sub)
    colnames(design) <- str_replace_all(colnames(design), fixed("as.factor(contrast)"), "")
    fit <- lmFit(tcga.fpkm[rownames(tcga.fpkm) %in% white.gene, pdata.sub$sample], design)
    contrast <- makeContrasts(c1_c2 = c1 - c2, levels = design)
    fits <- contrasts.fit(fit, contrast)
    ebFit <- eBayes(fits)
    deg_sig_list <- topTable(ebFit, coef = 1, adjust.method = 'fdr', number = Inf)
    deg.data <- deg_sig_list[which(!is.na(deg_sig_list$P.Value)), ]
    deg.data$logP <- -log10(deg.data$P.Value)
    deg.data$logFDR <- -log10(deg.data$adj.P.Val)
    deg.data$group = "zz"
    deg.data$group[which( (deg.data$adj.P.Val < 0.05) & (deg.data$logFC > 0.58) )] = "up"
    deg.data$group[which( (deg.data$adj.P.Val < 0.05) & (deg.data$logFC < -0.58) )] = "down"
    deg.data$tag <- paste0("TP53 mut -vs- wt")
    deg.data$Gene <- rownames(deg.data)
    deg.data$Cohort <- TCGA.ALL[kkk]
    write.xlsx(deg.data, paste0("output/degs_tp53_wtup_top/cohort/", TCGA.ALL[kkk], ".deg.xlsx" ), rowNames = T, overwrite = T)
    
    total.deg <- rbind(total.deg, deg.data)
    
    deg.data$Label = deg.data$Gene
    deg.data$Label[which(deg.data$group == "zz")] = ""
    p <- ggscatter(deg.data, 
                   x = "logFC", y = "logFDR",
                   color = "group", size = 0.5,
                   label = deg.data$Label, font.label = 4, repel = T,
                   main = paste0(TCGA.ALL[kkk], " TP53 mut vs wt"),
                   xlab = "log2FoldChange", ylab = "-log10(FDR)",
                   palette = c(down = "#00599F", up = "#D01910", zz = "#BBBBBB"))
    p <- p + theme_base()
    p <- p + geom_hline(yintercept = 1.30, linetype="dashed", color = "#222222")
    p <- p + geom_vline(xintercept = 0.58, linetype="dashed", color = "#222222")
    p <- p + geom_vline(xintercept = -0.58, linetype="dashed", color = "#222222")
    ggsave(paste0("output/degs_tp53_wtup_top/cohort/", TCGA.ALL[kkk], ".deg.pdf"), p, width = 9, height = 8)
  } else {
    message(TCGA.ALL[kkk])
  }
}

total.deg$Cohort <- factor(as.character(total.deg$Cohort), levels = TCGA.ALL)
saveRDS(total.deg, "output/degs_tp53_wtup_top/total.deg.rds")

colnames(plot.mat) <- TCGA.ALL
for (i in 1:ncol(plot.mat)) {
  sub <- plot.data[which(plot.data$TCGA.Cohort == colnames(plot.mat)[i]), ]

    plot.mat[, i] <- 0

}

stat.cohort <- stat.cohort[order(stat.cohort$SigCount, stat.cohort$NotSigCount, decreasing = T), ]

plot.mat <- plot.mat[, stat.cohort$Cohort]

              color = colorRampPalette(c("#fffef3","#edd1ff","#984ac9"))(501),

5.4 HR of high expression

HR of overexpression of p53 signature genes among TCGA cancer types.

5.4.1 116 p53 targets

Related to Fig. 3b.

plot.data <- read.xlsx("suppdata/TableS3-Related-to-Figure3A, 3B and S3G.xlsx", sheet = 7, startRow = 2)
gene.list <- unique(plot.data$Gene)

plot.data$HR_type <- 1
plot.data$HR_type[which(plot.data$HR < 0.67 | plot.data$HR > 1.5)] <- 1.5

plot.data$P_type <- "P ≥ 0.05"
plot.data$P_type[which(plot.data$P.value < 0.05 & plot.data$HR > 1)] <- "P < 0.05, HR > 1"
plot.data$P_type[which(plot.data$P.value < 0.05 & plot.data$HR < 1)] <- "P < 0.05, HR < 1"
plot.data$P_type <- factor(as.character(plot.data$P_type), levels = c("P < 0.05, HR < 1","P ≥ 0.05","P < 0.05, HR > 1"))


plot.data$Tag <- 0
plot.data$Tag[which(plot.data$Regulation.pattern.upon.ATO.treatment == "Downregulation (FC < 0.8) of unfavorable (HR > 1.5) genes")] <- 1
plot.data$Tag[which(plot.data$Regulation.pattern.upon.ATO.treatment == "Upregulation (FC > 1.25) of favorable (HR < 0.67) genes")] <- 1
plot.data$Tag[which(plot.data$Regulation.pattern.upon.ATO.treatment == "Downregulation (FC < 0.8) of favorable (HR < 0.67) genes")] <- -1
plot.data$Tag[which(plot.data$Regulation.pattern.upon.ATO.treatment == "Upregulation (FC > 1.25) of unfavorable (HR > 1.5) genes")] <- -1



plot.mat <- matrix(0, nrow = length(gene.list), ncol = length(TCGA.ALL))
rownames(plot.mat) <- gene.list
colnames(plot.mat) <- TCGA.ALL
for (i in 1:ncol(plot.mat)) {
  sub <- plot.data[which(plot.data$TCGA.Cohort == colnames(plot.mat)[i]), ]
  sub <- sub[which(sub$P.value < 0.05), ]
  plot.mat[match(sub$Gene, rownames(plot.mat)), i] <- log2(sub$HR)
}
stat.gene <- data.frame(Gene = rownames(plot.mat), 
                        SigCount = rowSums(plot.mat < 0) - rowSums(plot.mat > 0),
                        SigHR = rowMeans(plot.mat * -1) )
stat.gene <- stat.gene[order(stat.gene$SigCount, stat.gene$SigHR, decreasing = T), ]

stat.cohort <- data.frame(Cohort = colnames(plot.mat), 
                          SigCount = colSums(plot.mat < 0) - colSums(plot.mat > 0),
                          SigHR = colMeans(plot.mat * -1) )
stat.cohort <- stat.cohort[order(stat.cohort$SigCount, stat.cohort$SigHR, decreasing = T), ]

plot.info <- plot.data
plot.info$Cohort <- factor(as.character(plot.info$TCGA.Cohort), levels = stat.cohort$Cohort)
plot.info$Gene <- factor(as.character(plot.info$Gene), levels = rev(stat.gene$Gene) )

plot.info <- plot.info[order(plot.info$P_type, decreasing = T), ]
p <- ggplot(plot.info, aes(Cohort, Gene)) +
  geom_point(aes(size = HR_type, colour = P_type, fill = P_type), shape = 16, alpha = 0.7) + 
  xlab("") + ylab("")
p <- p + theme_base() + theme(plot.background = element_blank()) 
p <- p + scale_size(range = c(1.5,3))
p <- p + theme(axis.text.x = element_text(angle = 90, hjust = 1, vjust = 0.5, size = 12))
p <- p + theme(axis.text.y = element_text(size = 3))
p <- p + scale_colour_manual(values = c(`P < 0.05, HR > 1` = "#D01910", `P < 0.05, HR < 1` = "#00599F", `P ≥ 0.05` = "#CCCCCC"))
p
