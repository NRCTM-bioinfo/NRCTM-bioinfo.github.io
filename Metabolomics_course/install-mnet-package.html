Chapter 2 install MNet package

The R package MNet requires R version 4.0.0 or higher, please see https://cran.r-project.org.

相关依赖包安装

if(!require("BiocManager")){install.packages("BiocManager")}
library(BiocManager)
if(!require("Biobase")){BiocManager::install("Biobase")}
if(!require("BiocStyle")){BiocManager::install("BioStyle")}
if(!require("BiocGenerics")){BiocManager::install("BiocGenerics")}
if(!require("ropls")){BiocManager::install("ropls")}

if(!require("pheatmap")){install.packages("pheatmap")}
if(!require("pacman")){install.packages("pacman")}
library(pacman)
